% Custom title page command
% Supports 1-3 authors (separate with \and)
\newcommand{\inserttitlepage}{
    \begin{frame}[noframenumbering]
        \vspace{2em}
        \centering
        {\Huge \bfseries \inserttitle}

        \vspace{2em}
        % Authors - \and will create line breaks
        {\large \insertauthor}

        \vspace{1em}
        {\normalsize \insertinstitute}\\
        \vspace{0.5em}
        {\normalsize \insertdate}

        \vfill
    \end{frame}
}
